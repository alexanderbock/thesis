\newcommand{\ABpapermarker}[1]{%
  \sbox\ABpaperbox{%
    \fontfamily{lmss}\selectfont%
    {\fboxsep=10pt%
      \colorbox{niceblue}{\resizebox{!}{#1}{\color{white}\bfseries%\sffamily
        \Alph{pac}}%
    }}}%
  \rotatebox{90}{%
    \resizebox{%
      \heightof{\usebox{\ABpaperbox}}+\depthof{\usebox{\ABpaperbox}}}%
    {!}{%
      \sffamily%% \fontfamily{lmss}\selectfont%  %% enable this for sans serif
      \scshape\so \footnotesize Paper}}%
  \hspace{7pt}%\quad%  %% distance between vertikal "chapter" and numbered box
  \raisebox{\depthof{\usebox{\ABpaperbox}}}{\usebox{\ABpaperbox}}%
}

\newcommand{\palayout}[4]{ % #1=bibkey, #2=name
    \thispagestyle{empty}
    \hfill\makebox[2.5cm][r]{\ABpapermarker{2.5cm}}

    \begin{minipage}{0.9\linewidth}
        \hiddenlabel{pa:#2}{\Alph{pac}}
        \vspace{9cm}
    \begin{center}
        {\Large \citeti{#1}}\newline%
        \null
        ~\copyright~{#3}%~\usebibentry{#1}{publisher}
        \newline%
      \end{center}
      \bibentrywrap{#1}.
    \end{minipage}
}

     \iftoggle{isprint}{
         \cleartoevenpage[\thispagestyle{empty}]

     }
